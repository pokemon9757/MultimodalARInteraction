111 public void CompleteGrouping()
112 {
113 if (!_isInGroupMode)
114 {
115 Debug.LogWarning("Haven't started grouping, cannot complete group ");
116 return;
117 }
118 _isInGroupMode = false;
119
120 // Perform grouping
121 if (_objectsToGroupDict.Count == 0) return;
122 bool setParentPos = false;
123 Transform parent = new GameObject("Grouped Objects").transform;
124 foreach (InteractableObject obj in _objectsToGroupDict.Keys)
125 {
126 if (!obj.gameObject.activeSelf) continue;
127 obj.RecoverUpdatedColors();
128 Destroy(obj.GetComponent<InteractableObject>());
129
130 // Set the parent transform to the first child
131 if (!setParentPos)
132 {
133 parent.SetPositionAndRotation(obj.transform.position, obj.transform.rotation);
134 parent.localScale = obj.transform.localScale;
135 setParentPos = true;
136 }
137 obj.transform.parent = parent;
138 }
139 parent.gameObject.AddComponent<InteractableObject>();
140 _objectsToGroupDict.Clear();
141 _selectedObject = null;
142 _lastSelectedObject = null;

149 public void SelectObjectToGroup(bool active)
150 {
151 string action = active ? "select" : "deselect";
152
153 if (!_isInGroupMode)
154 {
155 Debug.LogError("Cannot perform " + action + ", not in group mode");
156 return;
157 }
158
159 // Take the last selected object if the currently selected is null
160 InteractableObject obj = _selectedObject ? _selectedObject : _lastSelectedObject;
161 if (obj == null)
162 {
163 Debug.LogWarning("Cannot perform " + action + " since no object has been recorded");
164 return;
165 }
166
167 // If selecting
168 if (active)
169 {
170 obj.UpdateColor(_selectedColor);
171 _objectsToGroupDict[obj] = active;
172 }
173 // Else if deselecting
174 else
175 {
176 // Recover to the original color if the dict contains the obj
177 if (_objectsToGroupDict.ContainsKey(obj))
178 {
179 obj.RecoverUpdatedColors();
180 _objectsToGroupDict.Remove(obj);
181 }
182 }
