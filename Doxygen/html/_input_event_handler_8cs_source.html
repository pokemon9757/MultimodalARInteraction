45 void OnCommandDetected(bool wasSuccessful, MLVoice.IntentEvent voiceEvent)
46 {
47 switch ((VoiceActions)voiceEvent.EventID)
48 {
49 case VoiceActions.Create:
50 // Get Shape and Color slots values
51 string shapeName = UtilityScript.GetSlotValue(voiceEvent.EventName, "Shape");
52 string colorName = UtilityScript.GetSlotValue(voiceEvent.EventName, "Color");
53 if (string.IsNullOrEmpty(shapeName))
54 {
55 Debug.LogError("Shape not found");
56 return;
57 }
58 if (string.IsNullOrEmpty(colorName))
59 colorName = "white";
60 Color? nullableColor = UtilityScript.StringToColor(colorName);
61 if (nullableColor == null)
62 {
63 Debug.LogWarning("The color is invalid, will not create a shape");
64 return;
65 }
66 // Perform explicit conversion from nullable Color to non-nullable Color
67 Color color = nullableColor ?? Color.white;
68 _createObjectAction.CreateNewObject(_eyeTracking.GazeMarkerPosition, color, shapeName);
69
70 break;
71
72 case VoiceActions.ChangeColor:
73 colorName = UtilityScript.GetSlotValue(voiceEvent.EventName, "Color");
74 nullableColor = UtilityScript.StringToColor(colorName);
75 if (nullableColor == null)
76 {
77 Debug.LogWarning("Invalid color " + colorName);
78 return;
79 }
80 color = nullableColor ?? (Color)nullableColor;
81 _interactorsManager.ChangeSelectedObjectColor(color);
82 break;
83
84 case VoiceActions.Delete:
85 DeleteObject();
86 break;
87
88 case VoiceActions.Group:
89 string action = UtilityScript.GetSlotValue(voiceEvent.EventName, "Action");
90 bool isStartAction = action == "Start" || action == "Begin";
91 if (isStartAction) _interactorsManager.StartGrouping();
92 else _interactorsManager.CompleteGrouping();
93 break;
94
95 case VoiceActions.Selection:
96 string selection = UtilityScript.GetSlotValue(voiceEvent.EventName, "Selection");
97 _interactorsManager.SelectObjectToGroup(selection == "Select");
98 break;
99
100 case VoiceActions.Scale:
101 string upDownValue = UtilityScript.GetSlotValue(voiceEvent.EventName, "UpDown");
102 bool isScaleUp = upDownValue == "Up" || upDownValue == "Bigger";
103 string percentageString = UtilityScript.GetSlotValue(voiceEvent.EventName, "Percentage");
104 int percent = Int32.Parse(percentageString);
105
106 _interactorsManager.ScaleSelectedObject(percent, isScaleUp);
107 break;
108 }
109 }
